### Task 1: Helper function `_get_skip_stages(state)`

In `commands.py`, add a private helper after `find_project_dir()`:

```python
import json as _json

def _get_skip_stages(state: ProjectState) -> set[str]:
    """Return set of stage names recorded as skipped in state metadata."""
    raw = state.metadata.get("skipped_stages", "[]")
    try:
        return set(_json.loads(raw))
    except (ValueError, TypeError):
        return set()
```

### Task 2: Implement fast mode logic in `plan()`

Replace the existing no-op fast mode section in `plan()` (`commands.py:447-499`):

**When `fast or skip_research_design`:**

```python
# --- FAST MODE PATH ---
import json as _json

# 1. State machine: synthetic transitions through skipped stages
#    (state machine only allows one step at a time)
sm_stage = Stage(state.current_stage)
# If we're at BRIEF, synthetically advance through RESEARCH → DESIGN
while sm_stage != Stage.DESIGN:
    if sm_stage == Stage.BRIEF:
        sm.transition(Stage.RESEARCH)
        sm_stage = Stage.RESEARCH
    elif sm_stage == Stage.RESEARCH:
        sm.transition(Stage.DESIGN)
        sm_stage = Stage.DESIGN
    else:
        break  # Already past DESIGN or at unexpected stage

# 2. Now transition to PLAN normally
if not sm.can_transition(Stage.PLAN):
    typer.echo(typer.style(f"Cannot enter plan from {sm.current_stage.value}", fg=typer.colors.RED))
    raise typer.Exit(code=1)

# 3. Preflight: only need BRIEF.md and brief_approved
skip_stages = {"research", "design"}
config = load_config(project_dir.parent)
check_preflight(Stage.PLAN, project_dir, skip_stages=skip_stages)

# 4. Build degraded context: tree + brief
brief_content = (project_dir / "BRIEF.md").read_text(encoding="utf-8")
codebase_tree = scan_codebase(project_dir.parent, config)
fast_research_sub = f"## Fast Mode: No research phase run\n\n{codebase_tree}"
fast_design_sub = '{"note": "Fast mode: no design phase run. Plan based on brief and codebase context."}'

# 5. LLM call
system_prompt, user_template = load_prompt("planner")
project_name = project_dir.parent.name
user_message = render_prompt(
    user_template,
    project_name=project_name,
    brief_content=brief_content,
    research_json=fast_research_sub,
    design_json=fast_design_sub,
)

typer.echo("Running planner (fast mode)...")
adapter = OpenAIAdapter(config)

def llm_call(prompt: str) -> str:
    response = adapter.call_for_json(system_prompt, prompt)
    return response.content

plan_data = validate_with_retry(llm_call, user_message, "plan", config, project_dir)
save_dual(plan_data, project_dir / "PLAN.json", project_dir / "PLAN.md")
typer.echo(typer.style("PLAN.json + PLAN.md saved.", fg=typer.colors.GREEN))

# 6. Approval gate
plan_md = (project_dir / "PLAN.md").read_text(encoding="utf-8")
approve_plan(state, project_dir / "STATE.json", plan_md)

# 7. Update state — set synthetic approvals, record skipped stages
state.approvals["research_approved"] = True
state.approvals["design_approved"] = True
sm.transition(Stage.PLAN)
state.current_stage = Stage.PLAN.value
state.metadata["skipped_stages"] = _json.dumps(sorted(skip_stages))
state.add_history("plan", "Fast mode: research and design skipped. Plan completed and approved.")
save_state(state, project_dir / "STATE.json")
typer.echo(typer.style("Plan approved. Stage: plan (fast mode)", fg=typer.colors.GREEN))
```
